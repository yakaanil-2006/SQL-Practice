Create table

CREATE TABLE users (
    id            INT AUTO_INCREMENT PRIMARY KEY,
    name          VARCHAR(100) NOT NULL,
    email         VARCHAR(100) UNIQUE NOT NULL,
    gender        ENUM('Male','Female','Other'),
    date_of_birth DATE,
    salary        INT,
    created_at    TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

ALTER TABLE users DROP COLUMN is_active;
ALTER TABLE users MODIFY COLUMN name VARCHAR(150);
ALTER TABLE users MODIFY COLUMN email VARCHAR(100) FIRST;
RENAME TABLE users TO customers;

INSERT — Adding Data

-- Specify columns (best practice)
INSERT INTO users (name, email, gender, date_of_birth, salary) VALUES
('Alice', 'alice@example.com', 'Female', '1995-05-14', 65000);

-- Multiple rows at once
INSERT INTO users (name, email, gender, date_of_birth, salary) VALUES
('Bob',     'bob@example.com',     'Male',   '1990-11-23', 70000),
('Charlie', 'charlie@example.com', 'Other',  '1988-02-17', 58000),
('Eva',     'eva@example.com',     'Female', '1993-12-30', 72000);

Always name columns — it protects against future table changes.

SELECT & WHERE

SELECT * FROM users;
SELECT name, email FROM users;

-- Comparisons
SELECT * FROM users WHERE salary > 60000;
SELECT * FROM users WHERE salary BETWEEN 50000 AND 70000;
SELECT * FROM users WHERE gender IN ('Male', 'Other');
SELECT * FROM users WHERE name LIKE 'A%';      -- starts with A
SELECT * FROM users WHERE name LIKE '%li%';   -- contains 'li'
SELECT * FROM users WHERE date_of_birth IS NULL;

-- Logical operators
SELECT * FROM users WHERE gender = 'Female' AND salary > 70000;
SELECT * FROM users WHERE gender = 'Male' OR gender = 'Other';

-- Sorting, pagination, distinct
SELECT * FROM users ORDER BY salary DESC;
SELECT * FROM users LIMIT 10 OFFSET 5;
SELECT DISTINCT gender FROM users;

UPDATE & DELETE

UPDATE users SET name = 'Alicia' WHERE id = 1;
UPDATE users SET name = 'Robert', email = 'r@example.com' WHERE id = 2;
UPDATE users SET salary = salary + 10000 WHERE salary < 60000;
DELETE FROM users WHERE id = 3;
DROP TABLE users;       -- removes table + data
TRUNCATE TABLE users;   -- clears rows, keeps structure, resets AUTO_INCREMENT

Always use WHERE with UPDATE/DELETE. Run SELECT first to confirm affected rows.

Functions

Aggregate

SELECT COUNT(*), MIN(salary), MAX(salary), SUM(salary), AVG(salary) FROM users;
SELECT gender, AVG(salary) AS avg_salary FROM users GROUP BY gender;

String

SELECT LENGTH(name), LOWER(name), UPPER(name) FROM users;
SELECT CONCAT(name, ' <', email, '>') AS contact FROM users;

Date

SELECT NOW();
SELECT name, YEAR(date_of_birth) FROM users;
SELECT name, DATEDIFF(CURDATE(), date_of_birth) AS days_lived FROM users;
SELECT name, TIMESTAMPDIFF(YEAR, date_of_birth, CURDATE()) AS age FROM users;

Math & conditional

SELECT ROUND(salary), FLOOR(salary), CEIL(salary) FROM users;
SELECT name, IF(gender = 'Female', 'Yes', 'No') AS is_female FROM users;

Transactions & AutoCommit

SET autocommit = 0;                          -- disable auto-save
UPDATE users SET salary = 80000 WHERE id = 5;
COMMIT;                                        -- save permanently
-- or
ROLLBACK;                                      -- undo since last COMMIT
SET autocommit = 1;                           -- re-enable

Foreign key with ON DELETE

CREATE TABLE addresses (
    id      INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    city    VARCHAR(100),
    CONSTRAINT fk_user FOREIGN KEY (user_id)
        REFERENCES users(id) ON DELETE CASCADE
);
ON DELETE optionBehavior
CASCADEAuto-deletes child rows
SET NULLSets FK to NULL in child
RESTRICTBlocks deletion if child exists (default)

JOINs

-- INNER JOIN — matching rows only
SELECT u.name, a.city FROM users u
INNER JOIN addresses a ON u.id = a.user_id;

-- LEFT JOIN — all users, NULL if no address
SELECT u.name, a.city FROM users u
LEFT JOIN addresses a ON u.id = a.user_id;

-- RIGHT JOIN — all addresses, NULL if no user
SELECT u.name, a.city FROM users u
RIGHT JOIN addresses a ON u.id = a.user_id;
TypeReturns
INNERRows that match in both tables
LEFTAll left rows + NULLs for no right match
RIGHTAll right rows + NULLs for no left match

UNION & UNION ALL

-- Removes duplicates
SELECT name FROM users
UNION
SELECT name FROM admin_users;

-- Keeps duplicates
SELECT name FROM users
UNION ALL
SELECT name FROM admin_users;

-- Add a role label
SELECT name, 'User' AS role FROM users
UNION
SELECT name, 'Admin' AS role FROM admin_users
ORDER BY name;

Column count and data types must match across all SELECT statements.

Self JOIN

Join a table with itself — useful for referral or hierarchical data.

ALTER TABLE users ADD COLUMN referred_by_id INT;

SELECT
  a.id,
  a.name AS user_name,
  b.name AS referred_by
FROM users a
LEFT JOIN users b ON a.referred_by_id = b.id;

Use aliases (a, b) to distinguish the two "copies" of the same table.

Views

A virtual table based on a saved SELECT — always reflects live data.

CREATE VIEW high_salary_users AS
SELECT id, name, salary FROM users WHERE salary > 70000;

SELECT * FROM high_salary_users;

DROP VIEW high_salary_users;

Indexes

SHOW INDEXES FROM users;

CREATE INDEX idx_email ON users(email);
CREATE INDEX idx_gender_salary ON users(gender, salary);
DROP INDEX idx_email ON users;

Indexes speed up reads but slow down writes. Only index columns used in WHERE, JOIN, or ORDER BY on large tables.

Subqueries

-- Scalar: users above average salary
SELECT id, name, salary FROM users
WHERE salary > (SELECT AVG(salary) FROM users);

-- IN: users referred by high earners
SELECT id, name FROM users
WHERE referred_by_id IN (
    SELECT id FROM users WHERE salary > 75000
);

-- In SELECT column
SELECT name, salary,
  (SELECT AVG(salary) FROM users) AS avg_salary
FROM users;

GROUP BY & HAVING

SELECT gender, COUNT(*) AS total FROM users GROUP BY gender;

-- HAVING filters after grouping (WHERE cannot use aggregates)
SELECT gender, AVG(salary) AS avg_salary
FROM users
GROUP BY gender
HAVING AVG(salary) > 75000;

-- ROLLUP adds subtotals + grand total
SELECT gender, COUNT(*) FROM users GROUP BY gender WITH ROLLUP;
ClauseRuns whenAggregates?
WHEREBefore groupingNo
GROUP BYGroups rows—
HAVINGAfter groupingYes

Stored Procedures

DELIMITER $$
CREATE PROCEDURE AddUser(
    IN p_name   VARCHAR(100),
    IN p_email  VARCHAR(100),
    IN p_gender ENUM('Male','Female','Other'),
    IN p_dob    DATE,
    IN p_salary INT
)
BEGIN
    INSERT INTO users (name, email, gender, date_of_birth, salary)
    VALUES (p_name, p_email, p_gender, p_dob, p_salary);
END$$
DELIMITER ;

CALL AddUser('Kiran', 'kiran@example.com', 'Female', '1994-06-15', 72000);
DROP PROCEDURE IF EXISTS AddUser;

Delimiter is changed to $$ so inner semicolons don't confuse MySQL. Always reset it after.

Triggers

CREATE TABLE user_log (
    id         INT AUTO_INCREMENT PRIMARY KEY,
    user_id    INT,
    name       VARCHAR(100),
    created_on TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

DELIMITER $$
CREATE TRIGGER after_user_insert
AFTER INSERT ON users
FOR EACH ROW
BEGIN
    INSERT INTO user_log (user_id, name)
    VALUES (NEW.id, NEW.name);
END$$
DELIMITER ;

DROP TRIGGER IF EXISTS after_user_insert;
KeywordMeaning
BEFORE / AFTERWhen to fire
NEW.colValue being inserted/updated
OLD.colValue before update/delete
FOR EACH ROWRun once per affected row
